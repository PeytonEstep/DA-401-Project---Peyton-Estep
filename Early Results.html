from textblob import TextBlob

# Sentiment Analysis

# creating function to get polarity - between -1 and 1
def get_sentiment(text):
    if pd.isna(text):
        return 0
    return TextBlob(text).sentiment.polarity

# applying it to trial 1-5 responses
df_early["Sentiment_Score"] = df_early["Response"].apply(get_sentiment)

# seeing if harmful responses are more positive than safe responses
print(df_early.groupby("Score")["Sentiment_Score"].mean())

Based on the above results, it's clear that ChatGPT is using the most "pleasant" language while failing its safety guardrails.

  • Score 1 = neutral - direct and prioritizing facts
  • Score 2 = lower positivity - AI is "confused" or giving contradictory advice
  • Score 3 = higher positivity - most pleasant language while failing safety protocols

So when the AI is harmful, it's over-validating the user.

from collections import Counter

def get_top_phrases(df_early, disorder, score, topic, n = 3):
    # filtering by score and topic
    subset = df_early[
        (df_early["Score"] == score) & 
        (df_early["Topic"] == topic) &
        (df_early["Disorder"] == disorder)]

    if subset.empty:
        return f"No examples found for {disorder} - {topic} with Score {score}"
    
    text = " ".join(subset["Response"].astype(str).lower())

    # finding phrases of "n" words
    words = re.findall(r'\b\w+\b', text)
    phrases = [" ".join(words[i:i+n]) for i in range(len(words)-n+1)]

    return Counter(phrases).most_common(10)
